class Solution:
    def findMaximumLength(self, nums: list[int]) -> int:
        n = len(nums)
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + nums[i]
        last = 0
        pos = 0
        length = 0
        while pos < n:
            lo, hi = pos, n
            while lo < hi:
                mid = (lo + hi) // 2
                if prefix[mid + 1] - prefix[pos] >= last:
                    hi = mid
                else:
                    lo = mid + 1
            if lo == n:
                break
            last = prefix[lo + 1] - prefix[pos]
            pos = lo + 1
            length += 1
        return length
