class Solution:
    def maximumStrength(self, nums: list[int], k: int) -> int:
        n = len(nums)
        INF = 10**18
        dp0 = [-INF] * (k + 1)
        dp1 = [-INF] * (k + 1)
        dp0[0] = 0
        for i in range(n):
            ndp0 = [-INF] * (k + 1)
            ndp1 = [-INF] * (k + 1)
            for j in range(k + 1):
                if dp0[j] != -INF:
                    ndp0[j] = max(ndp0[j], dp0[j])
                if dp1[j] != -INF:
                    ndp0[j] = max(ndp0[j], dp1[j])
                if j > 0:
                    coeff = k - j + 1
                    val = nums[i] * coeff
                    if dp0[j - 1] != -INF:
                        ndp1[j] = max(ndp1[j], dp0[j - 1] + val)
                    if dp1[j - 1] != -INF:
                        ndp1[j] = max(ndp1[j], dp1[j - 1] + val)
                    if dp1[j] != -INF:
                        ndp1[j] = max(ndp1[j], dp1[j] + val)
            dp0, dp1 = ndp0, ndp1
        return max(dp0[k], dp1[k])
